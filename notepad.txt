회원테이블(USER_INFO)
순서 / 아이디 / 비밀번호  / 이름  / 생년월일  / 성별   / 이메일 / 전화번호 / 가입한 날짜
no   / id    / password / name  / birthday / gender / email / Phone    / CreateDay
     / 5~20  / 4~18     / 0~40 /                   / 0~100 / 

CREATE TABLE IF NOT EXISTS pysql.USER_INFO (
    USER_NO BIGINT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    USER_ID VARCHAR(20) NOT NULL UNIQUE,
    USER_PASSWD VARCHAR(64) NOT NULL,
    USER_NAME VARCHAR(40) NOT NULL,
    USER_BIRTHDAY DATE NOT NULL,
    USER_GENDER ENUM('Male','Female') NOT NULL,
    USER_EMAIL VARCHAR(100) NULL,
    USER_PHONE VARCHAR(13) NOT NULL UNIQUE,
    USER_CREATEDAY DATETIME NOT NULL
) ENGINE = InnoDB DEFAULT CHARSET = utf8mb4;

메시지테이블(SAVE_MESSAGE)
순서 / 작성자 아이디 / 작성자 이름 / 내용 / 작성일
no  / id           / Name        / body / time

CREATE TABLE IF NOT EXISTS pysql.SAVE_MESSAGE(
    MESSAGE_NO BIGINT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    MESSAGE_ID VARCHAR(20) NOT NULL UNIQUE,
    MESSAGE_NAME VARCHAR(40) NOT NULL,
    MESSAGE_BODY VARCHAR(65532) NULL,
    MESSAGE_TIME DATETIME NOT NULL
) ENGINE = InnoDB;
